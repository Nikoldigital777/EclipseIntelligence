# Get Agent

> Retrieve details of a specific agent

## OpenAPI

````yaml openapi-final get /get-agent/{agent_id}
paths:
  path: /get-agent/{agent_id}
  method: get
  servers:
    - url: https://api.retellai.com
      description: The production server.
  request:
    security:
      - title: api key
        parameters:
          query: {}
          header:
            Authorization:
              type: http
              scheme: bearer
              description: >-
                Authentication header containing API key (find it in dashboard).
                The format is "Bearer YOUR_API_KEY"
          cookie: {}
    parameters:
      path:
        agent_id:
          schema:
            - type: string
              required: true
              description: Unique id of the agent to be retrieved.
              example: 16b980523634a6dc504898cda492e939
      query:
        version:
          schema:
            - type: integer
              required: false
              description: >-
                Optional version of the API to use for this request. If not
                provided, will default to latest version.
              example: 1
      header: {}
      cookie: {}
    body: {}

          import Retell from 'retell-sdk';


          const client = new Retell({

          const agentResponse = await
          client.agent.retrieve('16b980523634a6dc504898cda492e939');


          console.log(agentResponse.agent_id);
````
